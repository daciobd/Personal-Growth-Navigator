tsximport { View, Text, Pressable, StyleSheet, Platform } from 'react-native';
import { router, usePathname } from 'expo-router';
import { Feather } from '@expo/vector-icons';

const ITEMS = [
  { href: '/(tabs)', label: 'Hoje', icon: 'sun' },
  { href: '/(tabs)/meu-eu', label: 'Meu Eu', icon: 'user' },
  { href: '/journeys', label: 'Jornadas', icon: 'map' },
  { href: '/coach', label: 'Coach IA', icon: 'message-circle' },
];

export default function WebSidebar() {
  const pathname = usePathname();
  if (Platform.OS !== 'web') return null;
  return (
    <View style={styles.sidebar}>
      <Pressable onPress={() => router.replace('/(tabs)')} style={styles.logo}>
        <View style={styles.logoMark}>
          <Text style={styles.logoLetter}>J</Text>
        </View>
        <Text style={styles.logoName}>Jornada</Text>
      </Pressable>
      <View style={styles.nav}>
        {ITEMS.map(item => {
          const active = pathname === item.href || pathname.startsWith(item.href + '/');
          return (
            <Pressable
              key={item.href}
              style={[styles.item, active && styles.itemActive]}
              onPress={() => router.push(item.href)}>
              <Feather name={item.icon as any} size={18} color={active ? '#1B6B5A' : '#6B8F7E'} />
              <Text style={[styles.label, active && styles.labelActive]}>{item.label}</Text>
            </Pressable>
          );
        })}
      </View>
    </View>
  );
}

const styles = StyleSheet.create({
  sidebar: { position: 'fixed' as any, top: 0, left: 0, bottom: 0, width: 220, backgroundColor: '#fff', borderRightWidth: 1, borderRightColor: '#E8F0ED', paddingTop: 24, paddingHorizontal: 16, zIndex: 200 },
