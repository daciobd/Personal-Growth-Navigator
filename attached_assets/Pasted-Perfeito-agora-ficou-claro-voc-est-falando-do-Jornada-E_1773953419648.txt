# 🧩 O que o Jornada-EU precisa ter

Agora vem a resposta direta à sua pergunta.

## 1. Motor de execução (não só exibição)

O Jornada-EU NÃO pode só mostrar o plano.

Ele precisa:

* disparar no momento certo
* reduzir fricção
* facilitar o check-in em 1 toque

Exemplo:

* botão “feito” direto
* zero navegação

## 2. Sistema de check-in (core do app)

Esse é o coração:

* `done`
* `missed`
* opcional: `partial`

E cada ação precisa gerar evento:

```json
{
  "event": "done",
  "actionId": "a1",
  "timestamp": "...",
  "adaptationLevel": "minimal"
}
```

👉 Isso fecha o loop com o LONGEVI.

---

## 3. Loop de feedback em tempo real

O Jornada-EU precisa reagir:

Se usuário:

* falha → reduzir fricção (mais `minimal`)
* acerta → pode aumentar complexidade

Mas quem decide é o LONGEVI.
O Jornada-EU só **reflete rápido**.

---

## 4. UX orientada a hábito (não a informação)

Evitar:

* telas complexas
* múltiplas decisões

Focar em:

* 1 ação clara por vez
* linguagem simples
* progresso visível

Exemplo:

> “Agora: 1 copo de água 💧”

---

## 5. Persistência leve local

O Jornada-EU precisa guardar:

* plano atual
* estado das ações (done/missed)
* último sync

Para funcionar mesmo com latência/rede ruim.

Fluxo ideal:

1. App abre → busca plano
2. Usuário executa → gera eventos

# 2. Cliente de API

```ts
// /lib/jornada-api.ts

import type {
  DailyPlanRequest,
  DailyPlanResponse,
  BehavioralEventsRequest,
  BehavioralEventsResponse,
} from "@/types/jornada";

export async function fetchDailyPlan(
  input: DailyPlanRequest
): Promise<DailyPlanResponse> {
  const response = await fetch("/api/jornada/daily-plan", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(input),
  });

  if (!response.ok) {
    throw new Error("Failed to fetch daily plan");
  }

  return response.json();
}

export async function sendBehavioralEvents(
  input: BehavioralEventsRequest
): Promise<BehavioralEventsResponse> {
  const response = await fetch("/api/behavioral/events", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(input),
  });

  if (!response.ok) {
    throw new Error("Failed to send behavioral events");
  }

  return response.json();
}
```

# 4. Persistência local simples de fila

Para MVP, localStorage resolve.

```ts
// /lib/jornada-queue.ts

import type { BehavioralEvent } from "@/types/jornada";

const STORAGE_KEY = "jornada_pending_events";

export function getPendingEvents(): BehavioralEvent[] {
  if (typeof window === "undefined") return [];
  const raw = window.localStorage.getItem(STORAGE_KEY);
  if (!raw) return [];

  try {
    return JSON.parse(raw) as BehavioralEvent[];
  } catch {
    return [];
  }
}

export function enqueueEvent(event: BehavioralEvent) {
  const events = getPendingEvents();
  events.push(event);
  window.localStorage.setItem(STORAGE_KEY, JSON.stringify(events));
}

export function clearPendingEvents() {
  window.localStorage.removeItem(STORAGE_KEY);
}

export function replacePendingEvents(events: BehavioralEvent[]) {
  window.localStorage.setItem(STORAGE_KEY, JSON.stringify(events));
}
```

# 5. Flush da fila

```ts
// /lib/jornada-sync.ts

import { getPendingEvents, replacePendingEvents } from "@/lib/jornada-queue";
import { sendBehavioralEvents } from "@/lib/jornada-api";

export async function flushPendingEvents() {
  const pending = getPendingEvents();
  if (!pending.length) return;

  try {
    await sendBehavioralEvents({ events: pending });
    replacePendingEvents([]);
  } catch {
    // mantém para tentar depois
  }
}
```

```tsx
// /components/jornada/JornadaHome.tsx
"use client";

import { useEffect, useMemo, useState } from "react";
import type { DailyPlanResponse, PlanAction } from "@/types/jornada";
import { fetchDailyPlan, sendBehavioralEvents } from "@/lib/jornada-api";
import { buildBehavioralEvent } from "@/lib/jornada-events";
import { enqueueEvent } from "@/lib/jornada-queue";
import { flushPendingEvents } from "@/lib/jornada-sync";

type JornadaHomeProps = {
  userId: string;
  careMode: "adult" | "materna" | "pet";
};

type Status = "idle" | "done" | "missed";

export function JornadaHome({ userId, careMode }: JornadaHomeProps) {
  const [plan, setPlan] = useState<DailyPlanResponse | null>(null);
  const [loading, setLoading] = useState(true);
  const [status, setStatus] = useState<Status>("idle");
  const [error, setError] = useState<string | null>(null);

  const primaryAction = useMemo<PlanAction | null>(() => {
    if (!plan?.actions?.length) return null;
    return [...plan.actions].sort((a, b) => a.priority - b.priority)[0];
  }, [plan]);

  useEffect(() => {
    void load();
    void flushPendingEvents();
  }, []);

  async function load() {
    try {
      setLoading(true);
      setError(null);

      const nextPlan = await fetchDailyPlan({
        userId,
        careMode,
        context: {
          timeOfDay: getTimeOfDay(),
          dayOfWeek: getDayOfWeek(),
          timezone: Intl.DateTimeFormat().resolvedOptions().timeZone,
          surface: "home",
        },
      });

      setPlan(nextPlan);

      const loadedEvent = buildBehavioralEvent({
        eventType: "plan_loaded",
        userId,
        plan: nextPlan,
        action: null,
        surface: "home",
      });

      await safeSendEvent(loadedEvent);
    } catch (err) {
      setError("Não foi possível carregar seu plano de hoje.");
    } finally {
      setLoading(false);
    }
  }

  async function safeSendEvent(event: ReturnType<typeof buildBehavioralEvent>) {
    try {
      await sendBehavioralEvents({ events: [event] });
    } catch {
      enqueueEvent(event);
    }
  }

  async function handleDone() {
    if (!plan || !primaryAction) return;

    const event = buildBehavioralEvent({
      eventType: "done",
      userId,
      plan,
      action: primaryAction,
      surface: "home",
      extraMetadata: {
        completionMethod: "one_tap",
      },
    });

    setStatus("done");
    await safeSendEvent(event);
  }

  async function handleMissed() {
    if (!plan || !primaryAction) return;

    const event = buildBehavioralEvent({
      eventType: "missed",
      userId,
      plan,
      action: primaryAction,
      surface: "home",
      extraMetadata: {
        reason: "not_completed",
      },
    });

    setStatus("missed");
    await safeSendEvent(event);
  }

  if (loading) {
    return (
      <main className="mx-auto max-w-md p-6">
        <div className="rounded-3xl border p-6 shadow-sm">
          <p className="text-sm text-neutral-500">Carregando seu dia...</p>
        </div>
      </main>
    );
  }

  if (error || !plan || !primaryAction) {
    return (
      <main className="mx-auto max-w-md p-6">
        <div className="rounded-3xl border p-6 shadow-sm space-y-3">
          <h1 className="text-xl font-semibold">Jornada-EU</h1>
          <p className="text-sm text-neutral-600">
            {error ?? "Nenhuma ação disponível para hoje."}
          </p>
          <button
            onClick={() => void load()}
            className="rounded-2xl border px-4 py-2 text-sm"
          >
            Tentar novamente
          </button>
        </div>
      </main>
    );
  }

  return (
    <main className="mx-auto max-w-md p-6">
      <div className="rounded-3xl border p-6 shadow-sm space-y-6">
        <header className="space-y-1">
          <p className="text-sm text-neutral-500">{getGreeting()}</p>
          <h1 className="text-2xl font-semibold">Sua jornada de hoje</h1>
        </header>

        {status === "idle" && (
          <ActionCard plan={plan} action={primaryAction} />
        )}

        {status === "done" && (
          <FeedbackCard
            title="Feito 👏"
            description="Você concluiu sua ação de hoje."
          />
        )}

        {status === "missed" && (
          <FeedbackCard
            title="Tudo bem."
            description="Vamos tentar de forma mais simples amanhã."
          />
        )}

        {status === "idle" && (
          <div className="flex gap-3">
            <button
              onClick={() => void handleDone()}
              className="flex-1 rounded-2xl px-4 py-3 font-medium border"
            >
              Feito
            </button>
            <button
              onClick={() => void handleMissed()}
              className="flex-1 rounded-2xl px-4 py-3 text-neutral-600 border"
            >
              Não consegui
            </button>
          </div>
        )}

        {plan.uiHints?.showProgress && (
          <p className="text-xs text-neutral-500">
            Estratégia do dia: {plan.adaptation.level}
          </p>
        )}
      </div>
    </main>
  );
}

function ActionCard({
  plan,
  action,
}: {
  plan: DailyPlanResponse;
  action: PlanAction;
}) {
  const isMinimal = plan.adaptation.level === "minimal";
  const isSimplified = plan.adaptation.level === "simplified";

  return (
    <section className="space-y-3">
      <div className="rounded-2xl bg-neutral-50 p-4">
        <p className="text-sm text-neutral-500">Ação principal</p>
        <h2 className="text-xl font-medium mt-1">{action.title}</h2>
        <p className="text-sm text-neutral-700 mt-2">{action.instruction}</p>

        {isSimplified && (
          <p className="text-sm text-neutral-500 mt-3">
            Pequenas ações consistentes valem mais do que planos perfeitos.
          </p>
        )}

        {!isMinimal && plan.adaptation.level === "normal" && (
          <div className="mt-4 space-y-1 text-sm text-neutral-500">
            <p>Tipo: {action.type}</p>
            {action.trigger ? <p>Momento: {action.trigger}</p> : null}
            <p>Esforço: {action.effort}</p>
          </div>
        )}
      </div>
    </section>
  );
}

function FeedbackCard({
  title,
  description,
}: {
  title: string;
  description: string;
}) {
  return (
    <section className="rounded-2xl bg-neutral-50 p-4 space-y-2">
      <h2 className="text-xl font-medium">{title}</h2>
      <p className="text-sm text-neutral-600">{description}</p>
    </section>
  );
}

function getGreeting() {
  const hour = new Date().getHours();
  if (hour < 12) return "Bom dia";
  if (hour < 18) return "Boa tarde";
  return "Boa noite";
}

function getTimeOfDay() {
  const hour = new Date().getHours();
  if (hour < 12) return "morning";
  if (hour < 18) return "afternoon";
  return "evening";
}

function getDayOfWeek() {
  return new Intl.DateTimeFormat("en-US", { weekday: "long" })
    .format(new Date())
    .toLowerCase();
}
```
