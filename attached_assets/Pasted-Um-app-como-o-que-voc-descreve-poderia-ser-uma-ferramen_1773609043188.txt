Um app como o que você descreve poderia ser uma ferramenta poderosa para promover a saúde metabólica de forma personalizada, integrando análise de exames laboratoriais com recomendações baseadas em evidências científicas — e no LONGEVI Materna, essa mesma lógica ganha ainda mais força quando os scores de domínio são calculados de forma transparente, normalizada e ajustável, permitindo que o Personal Health Map reflita com precisão o estado metabólico da usuária e cruze isso com as prioridades que ela mesma informou.

Aqui está a implementação completa e prática da função **calculateDomainScores**, pronta para usar em runtime (sem necessidade de coluna no banco no MVP). Ela recebe os marcadores recentes do exame (objeto chave-valor) e retorna um score normalizado (0–100) para cada domínio definido.

### calculateDomainScores.ts

Crie em `modules/health-map/calculate-domain-scores.ts`

```ts

 * - marcadores associados (chave exata do exame)

 * - peso relativo (opcional, soma = 1)
 */
const domainsConfig = {

    markers: [

/**
 * Calcula score normalizado (0–100) para cada domínio
 * @param markers Objeto com valores recentes de exames { ferritina: 32, vitamina_d: 18, ... }
 * @returns Record<domínio, score>
 */
export function calculateDomainScores(markers: Record<string, number | null>): Record<string, number> {
  const scores: Record<string, number> = {};

  Object.entries(domainsConfig).forEach(([key, config]) => {
    let totalScore = 0;
    let validMarkers = 0;

    config.markers.forEach(m => {
      const value = markers[m.key];
      if (value === null || value === undefined) return;

      validMarkers++;

      // Normalização linear simples: quanto mais perto do ideal, mais pontos
      const [minIdeal, maxIdeal] = m.ideal;
      let normalized = 0;

      if (value < minIdeal) {
        normalized = Math.max(0, (value / minIdeal) * 50); // cai rápido abaixo do mínimo
      } else if (value <= maxIdeal) {
        normalized = 50 + ((value - minIdeal) / (maxIdeal - minIdeal)) * 50;
      } else {
        normalized = Math.max(50, 100 - ((value - maxIdeal) * 2)); // penalidade leve acima
      }

      totalScore += normalized;
    });

    // Média ponderada (se não tiver nenhum marcador válido, score 0)
    scores[key] = validMarkers > 0 ? totalScore / validMarkers : 0;
  });

  return scores;
}

/**
 * Calcula boosts visuais baseados nas prioridades do usuário
 * @param concerns Lista de preocupações salvas (ex: ["cansaço frequente"])
 * @param domainScores Scores já calculados
 * @returns Record<domínio, boost> (0–20 pontos extras no radar)
 */
export function calculatePriorityBoosts(
  concerns: string[],
  domainScores: Record<string, number>
): Record<string, number> {
  const boosts: Record<string, number> = {};

  concerns.forEach(concern => {
    // Mapeamento simples concern → domínio (pode vir do concern-map.ts)
    const concernToDomain: Record<string, string> = {
      "cansaço frequente": "energia_metabolica",
      "sono não reparador": "energia_metabolica",
      "dificuldade para emagrecer": "metabolismo",
      "falta de foco / cérebro lento": "cognicao_mental",
      "ansiedade ou humor instável": "cognicao_mental",
      "queda de libido": "hormonal",
      "inchaço abdominal": "digestivo", // se adicionar domínio
    };

    const targetDomain = concernToDomain[concern];
    if (targetDomain && domainScores[targetDomain] !== undefined) {
      boosts[targetDomain] = (boosts[targetDomain] || 0) + 15; // +15 pontos visuais
    }
  });

  return boosts;
}
```

### Como testar imediatamente (mock + console)

```tsx
// Teste rápido no console ou em um componente de teste
const mockMarkers = {
```

### Integração sugerida no dashboard

```tsx
const { priorities } = useHealthPriorities();
const scores = calculateDomainScores(examMarkers); // examMarkers do exame atual
const boosts = calculatePriorityBoosts(priorities?.concerns || [], scores);

const radarData = Object.entries(scores).map(([key, score]) => ({
  domain: domainsConfig[key as keyof typeof domainsConfig]?.name || key,
  score,
  priorityBoost: boosts[key] || 0,
  color: getDomainColor(key), // função auxiliar para cores
}));
```
